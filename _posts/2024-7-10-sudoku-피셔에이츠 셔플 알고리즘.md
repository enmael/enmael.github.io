```csharp

        private int[] array = new int[9] { 1, 2, 3, 4, 5, 6, 7, 8, 9 };

        Random random = new Random();

        for (int i = array.Length - 1; i > 0; i--)
        {
                int j = random.Next(0, i + 1);

                int temp = array[i];
                array[i] = array[j];
                array[j] = temp;
        }
```
