```csharp
   public class AdjacencyList<T>
    {
        private class Node
        {
            public T data;
            public Node next;

            public Node(T data, Node link = null)
            {
                this.data = data;
                next = link;
            }



        }

        private int size;
        private int arraySize;

        private T[] vertex;
        private Node[] list;

        public AdjacencyList()
        {
            size = 0;
            arraySize = 10;

            vertex = new T[arraySize];
            list = new Node[arraySize];
        }

        public void Insert(T data)
        {
            if (size <= arraySize)
            {
                vertex[size] = data;
                size++;
            }
        }

        public void Connect(int u, int v)
        {
            if (size <= 0)
            {
                Console.WriteLine("Adjancency list is empty");
                return;
            }

            if (u >= size || v >= size)
            {
                Console.WriteLine("");
                return;
            }
            list[u] = new Node(vertex[v], list[u]);
            list[v] = new Node(vertex[u], list[v]);

        }

        public void Show()
        {
            for (int i = 0; i < size; i++)
            {
                Console.Write(vertex[i] + " : ");

                Node currentNode = list[i];

                while (currentNode != null)
                {
                    Console.Write(currentNode.data + " ");

                    currentNode = currentNode.next;
                }
                Console.WriteLine();

            }
        }
    }
    #endr
```
